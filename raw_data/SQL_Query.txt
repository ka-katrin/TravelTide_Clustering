--Data is supposed to be filtered for active Customers-> only the ones, that have had at least 8 Sessions after the 01.04.23

-- session filter for more than 7 sessions  after the 05.01.23
with active_users as (
  select user_id, count(session_id)
  from sessions
  where session_start >= '2023-01-05'
  group by user_id
  having count(session_id)> 7
  )
-- only active users in users table -> filtered to download
select *
from users
where user_id in (select user_id from active_users)

--flights table - only flights, that are valid from active users (but also data form before 2023 from active users)

with active_users as (
  select user_id, count(session_id)
  from sessions
  where session_start >= '2023-01-05'
  group by user_id
  having count(session_id)> 7
  ),

active_trips as (
  select trip_id
  from sessions
  where user_id in (select user_id from active_users))

select *
from flights
where trip_id in (select trip_id from active_trips)
